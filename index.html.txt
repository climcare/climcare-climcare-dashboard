    <button class="btn reset" onclick="enviarReset()">Resetar Wi-Fi</button>

    <script>
        let caracteristica;

        async function conectar() {
            try {
                const dispositivo = await navigator.bluetooth.requestDevice({
                    filters: [{ name: 'ClimCare-Diag' }],
                    optionalServices: ['181a']
                });
                const servidor = await dispositivo.gatt.connect();
                const servico = await servidor.getPrimaryService('181a');
                caracteristica = await servico.getCharacteristic('2a58');

                // Ativa atualizações automáticas
                await caracteristica.startNotifications();
                caracteristica.addEventListener('characteristicvaluechanged', (event) => {
                    const decoder = new TextDecoder('utf-8');
                    const texto = decoder.decode(event.target.value);
                    // Lógica simples para extrair T e U da string
                    const temp = texto.match(/T: ([\d.]+)/);
                    const umid = texto.match(/U: ([\d.]+)/);
                    if(temp) document.getElementById('temp').innerText = temp[1] + "°C";
                    if(umid) document.getElementById('umid').innerText = umid[1] + "%";
                });
                alert("Conectado com sucesso!");
            } catch (erro) { console.error(erro); }
        }

        async function enviarReset() {
            if (!caracteristica) return alert("Conecte primeiro!");
            const encoder = new TextEncoder();
            await caracteristica.writeValue(encoder.encode("RESET"));
            alert("Comando de Reset enviado!");
        }
    </script>
